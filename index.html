// 1. byte, char, short 类型数据进行计算时，计算结果自动转换为int型
// 2. 小数常量自动为double类型
public class HelloWrold{
    public static void main(String[] args){
        char A = 'b' - 'a';    //error
        int  B = 'b' - 'a';    //B = 1
        float C = 1.05;        //error
        float D = 1.05F;       //D = 1.05
    }
}

// 2. 若表达式中出现不同的转换类型，则记过自动转换成两者中范围更大的数据类型
